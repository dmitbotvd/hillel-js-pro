Примеры с привидением типа

40 / "4" = 4

true + true = 2

"valera" + 123 = valera123

+"true" = NaN

20 > '40' = false

null >= 0 = true

null == undefined = true

null === undefined = false

true + 40 * 2 = 81

'valera' >= 'valer' = true

true || false = true

true && true && 'valera' = 'valera'

false || 0 || '' = ''
